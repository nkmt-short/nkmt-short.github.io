function Boundary(omega, psi, h)
    global omega
    omega[1,:]  =-2.0/h^2 .*psi[2,:]
    omega[N+1,:]=-2.0/h^2 .*psi[N,:]
    omega[:,1]  =-2.0/h^2 .*psi[:,2]
    omega[:,N+1]=-2.0/h^2 .*(psi[:,N].+h)
end

function NewOmega(om, psi, Re)
    for i=2:N
        for j=2:N
            om[i,j]=(om[i+1,j]+om[i-1,j]+om[i,j+1]+om[i,j-1])/4.0 + 
                Re/16.0 * ( (psi[i+1,j]-psi[i-1,j])*(om[i,j+1]-om[i,j-1]) - (om[i+1,j]-om[i-1,j])*(psi[i,j+1]-psi[i,j-1]) )
        end
    end
end

function NewPsi(om, psi, h)
    for i=2:N
        for j=2:N
            psi[i,j]=(psi[i+1,j]+psi[i-1,j]+psi[i,j+1]+psi[i,j-1])/4.0 + h^2/4.0*om[i,j]
        end
    end
end

N=100
h=1.0/N
omega=zeros(N+1,N+1)
psi  =zeros(N+1,N+1)
Re=100

for step=1:1500
    Boundary(omega, psi, h)
    NewOmega(omega, psi, Re)
    NewPsi(omega, psi, h)
end
